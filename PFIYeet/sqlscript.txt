CREATE TABLE IF NOT EXISTS usager
(usagerID INTEGER(10) AUTO_INCREMENT PRIMARY KEY,
username VARCHAR(40) UNIQUE NOT NULL,
email VARCHAR(40) NOT NULL,
password VARCHAR(250), 
imageProfil VARCHAR(100) NOT NULL);

CREATE TABLE IF NOT EXISTS album
(albumID INTEGER(10) AUTO_INCREMENT PRIMARY KEY,
titre VARCHAR(40) UNIQUE NOT NULL,
proprietaire VARCHAR(40) NOT NULL,
description VARCHAR(40) NOT NULL,
tempsCreation Date NOT NULL);

CREATE TABLE IF NOT EXISTS image
(imageID INTEGER AUTO_INCREMENT PRIMARY KEY,
URL VARCHAR(100) NOT NULL,
albumID INTEGER NOT NULL,
description VARCHAR(100) NOT NULL,
tempsCreation Date NOT NULL);

CREATE TABLE IF NOT EXISTS commentaire
(commentaireID 	INTEGER(10) AUTO_INCREMENT PRIMARY KEY,
typeObjet VARCHAR(10) NOT NULL,
tempsCreation Date NOT NULL,
contenu VARCHAR(200) NOT NULL);

CREATE TABLE IF NOT EXISTS likes
(
ojbectID INTEGER(10) AUTO_INCREMENT PRIMARY KEY,
usagerID INTEGER(10) NOT NULL,
typeObject VARCHAR(20) NOT NULL
)
